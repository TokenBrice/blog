{{- /* BreadcrumbList JSON-LD. Most posts use `url:` overrides (flat /slug/) which collapses .Ancestors and hides the category step. We emit an explicit chain instead: Home -> first category (if any) -> page. For non-post pages with real ancestors, we still emit ancestor chain. */ -}} {{- if not .IsHome -}} {{- $items := slice -}} {{- /* Home */ -}} {{- $items = $items | append (dict "name" .Site.Title "url" .Site.BaseURL) -}} {{- /* Category step for posts: first item in .Params.categories */ -}} {{- $usedCategory := false -}} {{- if .IsPage -}} {{- with .Params.categories -}} {{- $first := "" -}} {{- if reflect.IsSlice . -}} {{- $first = index . 0 -}} {{- else -}} {{- $first = . -}} {{- end -}} {{- with $first -}} {{- /* Use absLangURL so FR posts get tokenbrice.xyz/fr/categories/X/ */ -}} {{- $catURL := absLangURL (printf "categories/%s/" (. | urlize)) -}} {{- $items = $items | append (dict "name" . "url" $catURL) -}} {{- $usedCategory = true -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Otherwise use Ancestors when present (sections, taxonomies, etc.) */ -}} {{- if not $usedCategory -}} {{- range .Ancestors.Reverse -}} {{- if not .IsHome -}} {{- $items = $items | append (dict "name" .Title "url" .Permalink) -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Current page */ -}} {{- $items = $items | append (dict "name" .Title "url" .Permalink) -}} {{- end -}}
